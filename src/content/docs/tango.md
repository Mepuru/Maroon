let lesson = "all";
let mode = "both";

function render() {
  const filtered = lesson === "all" ? WORDS : WORDS.filter(w => w.lesson === lesson);
  let html = "";
  for (let i = 0; i < filtered.length; i++) {
    const w = filtered[i];
    const cn = mode === "kana"
      ? '<span class="dim">' + w.kanji + '</span>'
      : '<span>' + w.kanji + '</span>';
    const kn = mode === "kanji"
      ? '<span class="dim">' + w.kana + '</span>'
      : '<span>' + w.kana + '</span>';
    const n = i + 1;
    html += '<tr><td class="num">' + n + '</td>'
         + '<td class="kanji">' + cn + '</td>'
         + '<td class="kana">' + kn + '</td>'
         + '<td class="lesson-label">' + w.lesson + '</td></tr>';
  }
  document.getElementById("tbody").innerHTML = html;
  document.getElementById("count").textContent = filtered.length + " / " + WORDS.length + " 词";
}

function filterLesson(val) { lesson = val; render(); }

function setMode(m) {
  mode = m;
  document.querySelectorAll(".mode-btn").forEach(b => b.classList.toggle("active", b.dataset.mode === m));
  render();
}
